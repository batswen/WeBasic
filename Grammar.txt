program:    statement (lf statement)*
statement:  variable "=" orexpr


orexpr:     andexpr ("OR" andexpr)*
andexpr:    eqexpr ("AND" eqexpr)*
eqexpr:     glexpr (("=="|"!=") eqexpr)*
glexpr:     expr (("<"|"<="|">"|">=") expr)*

expr:       term (("+"|"-") term)*
term:       factor (("*"|"/") factor)*
factor:      ("+"|"-") number
            '"' chars '"'
            "(" expr ")"
            variable

number:     [0-9] ("." [0-9]*)
variable:   [a-zA-Z_\$][\$_0-9\p{L}]*
lf:         (":"|"\n")
